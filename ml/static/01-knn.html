'''It\'s the time for us to use a classifier ; knn seems to be the most suitable model in this case ! We can use
cross validation and some model selection to improve knn results'''
from sklearn.neighbors import KNeighborsClassifier
from sklearn.model_selection import cross_val_score, cross_val_predict, LeaveOneOut

# first prepare our features and target
X = df.drop('target', axis='columns')
y = df['target']

# select our cross validation iterator
cvi = LeaveOneOut()

# model selection, determine the best k-neighbors value for the set
scores = []
ks = [x for x in range(0,30) if(x%2!=0)]
for k in ks:
    # create classifier
    knn = KNeighborsClassifier(n_neighbors=k)
    # cross validation with 5 folds
    a = cross_val_score(knn, X, y, cv=cvi)
    scores.append(a)
